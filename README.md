- node_modules
- src
  - util
    - module-alias.ts
- test
  - ass
- .gitignore
- package.json
- tsconfig.json

## steps settings

- config .gitignore
- init project with: npm init or npm init -y
- install dependency typescript with: yarn add -D typescript
- install dependency types with: yarn add -D @types/node
- install dependency alias with: yarn add module-alias
- install dependency types with: yarn add -D @types/node
